Bitwise operations

a = 60  # 60 = 0011 1100
b = 13  # 13 = 0000 1101
c = 0

# 12 = 0000 1100

# 61 = 0011 1101
print("line 2 - Value of c is ", c)

c = a ^ b
# 49 = 0011 0001
print("line 3 - Value of c is ", c)

c = ~a
# -61 = 1100 0011
print("line 4 - Value of c is ", c)

c = a << 2
# 240 = 1111 0000
print("line 5 - Value of c is ", c)

c = a >> 2
# 15 = 0000 1111

Operators Precedence

a = 20
b = 10
c = 15
d = 5
e = 0

e = (a + b) * c / d  # ( 30 * 15 ) / 5
print("value of (a + b) * c / d is ", e)

e = ((a + b) * c) / d  # (30 * 15 ) / 5
print("value of ((a + b) * c) / d is ", e)

e = (a + b) * (c / d)
# (30) * (15/5)
print("value of (a + b) * (c / d) is ", e)

e = a + (b * c) / d
#  20 + (150/5)
print("value of a + (b * c) / d is ", e)

value of (a + b) * c / d is  90.0
value of ((a + b) * c) / d is  90.0
value of (a + b) * (c / d) is  90.0
value of a + (b * c) / d is  50.0
