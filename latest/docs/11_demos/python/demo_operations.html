Operators Precedence

a = 20
b = 10
c = 15
d = 5
e = 0

e = (a + b) * c / d  # ( 30 * 15 ) / 5
print("value of (a + b) * c / d is ", e)

e = ((a + b) * c) / d  # (30 * 15 ) / 5
print("value of ((a + b) * c) / d is ", e)

e = (a + b) * (c / d)
# (30) * (15/5)
print("value of (a + b) * (c / d) is ", e)

e = a + (b * c) / d
#  20 + (150/5)
print("value of a + (b * c) / d is ", e)

value of (a + b) * c / d is  90.0
value of ((a + b) * c) / d is  90.0
value of (a + b) * (c / d) is  90.0
value of a + (b * c) / d is  50.0
