        mplcyberpunk.make_lines_glow()
        mplcyberpunk.add_gradient_fill(alpha_gradientglow=0.5, gradient_start="zero")

        plt.xlabel("X")
        plt.ylabel("Y")
        plt.grid()
        plt.show()
