345 static inline void parseUiDescription(int uiDescription, int* uiType = NULL, int* buttonBarType = NULL, bool* childOfMainWindow = NULL, bool* deleteOnClose = NULL, int* dockWidgetArea = NULL)
346 {
347 if(uiType) *uiType = (uiDescription & 0x000000FF); //bits 1-8
348 if(buttonBarType) *buttonBarType = ((uiDescription & 0x0000FF00) >> 8); //bits 9-16
349 if(childOfMainWindow) *childOfMainWindow = ((uiDescription & 0x000F0000) > 0); //bits 17-20
350 if(deleteOnClose) *deleteOnClose = ((uiDescription & 0x00F00000) > 0); //bits 21-24
351 if(dockWidgetArea) *dockWidgetArea = ((uiDescription & 0xFF000000) >> 24); //bits 25-32
352 }
353
354 static inline int createUiDescription(int uiType, int buttonBarType, bool childOfMainWindow, bool deleteOnClose, int dockWidgetArea)
355 {
356 int v = uiType & 0x000000FF; //bits 1-8
357 v += (buttonBarType << 8); //bits 9-16
358 if(childOfMainWindow) v += (1 << 16); //bits 17-24
359 if(deleteOnClose) v+= (1 << 20); //bits 21-24
360 v += (dockWidgetArea << 24); //bits 25-32
361
362 return v;
363 }
