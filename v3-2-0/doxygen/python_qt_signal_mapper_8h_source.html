78  PythonQtSignalTarget(IntList &argTypeList, int slotId, int signalId, PyObject* callable, const char *signal, int minRepeatInterval) :
79  m_slotId(slotId),
80  m_signalId(signalId),
81  m_function(NULL),
82  m_boundedInstance(NULL),
83  m_boundedMethod(false),
84  m_signalName(signal),
85  m_minRepeatInterval(minRepeatInterval)
86  {
87  m_argTypeList = argTypeList;
88  PyObject *temp = NULL;
89  m_elapsedTimer.invalidate();
90 
91  if(PyMethod_Check(callable))
92  {
93  m_boundedMethod = true;
94  Py_XDECREF(m_boundedInstance);
95  Py_XDECREF(m_function);
96  temp = PyMethod_Self(callable); //borrowed
97  m_boundedInstance = PyWeakref_NewRef(temp, NULL); //new ref (weak reference used to avoid cyclic garbage collection)
98  temp = PyMethod_Function(callable); //borrowed
99  m_function = PyWeakref_NewRef(temp, NULL); //new ref
100  }
101  else if(PyFunction_Check(callable))
102  {
103  m_boundedMethod = false;
104  Py_XDECREF(m_boundedInstance);
105  Py_XDECREF(m_function);
106  m_function = PyWeakref_NewRef(callable, NULL); //new ref
107  }
108  };

111  PythonQtSignalTarget(const PythonQtSignalTarget &copy) :
112  m_slotId(-1),
113  m_signalId(-1),
114  m_function(NULL),
115  m_boundedInstance(NULL),
116  m_boundedMethod(false),
117  m_signalName(copy.m_signalName)
118  {
119  Py_XDECREF(m_boundedInstance);
120  Py_XDECREF(m_function);
121  m_slotId = copy.slotId();
122  m_signalId = copy.signalId();
123  m_argTypeList = copy.argTypeList();
124 
125  m_boundedMethod = copy.m_boundedMethod;

128  m_boundedInstance = copy.m_boundedInstance;
129  Py_XINCREF(m_boundedInstance);

133  }

163  bool isSame(int signalId, PyObject* callable) const
164  {
165  if(signalId == m_signalId)
166  {
167  if(PyMethod_Check(callable))
168  {
169  return PyMethod_Self(callable) == PyWeakref_GetObject(m_boundedInstance) && PyMethod_Function(callable) == PyWeakref_GetObject(m_function);
170  }
171  return callable == PyWeakref_GetObject(m_function);
172  }
173  return false;
174  }
175

177  int m_slotId;

181  PyObject *m_boundedInstance;
182  bool m_boundedMethod;
183  QString m_signalName;
184  QElapsedTimer m_elapsedTimer;
185 
186  /*
187  if > 0, every call of a certain slot by a certain signal with restart m_elapsedTimer. If the same signal-slot-connection is
188  called another time, while less than m_minRepeatInterval ms have expired, this new invoke will be ignored.
189 
190  This can be used to prevent that very fast signal emissions will overflow the call queue for the connected callable python method.
191  */
192  int m_minRepeatInterval;
