198  ShortcutAction(const QIcon &icon, const QString &text, AbstractDockWidget *parent, const QKeySequence &key, Qt::ShortcutContext context = Qt::WindowShortcut) : QObject(parent), m_action(NULL), m_shortcut(NULL)
199  {
200  QString text2 = text;
201  QString text3 = text;
202  text3.replace("&", "");
203 
204  //some key sequences do not exist as default on all operating systems
205  if (key.isEmpty() == false)
206  {
207  text2 += "\t" + key.toString(QKeySequence::NativeText);
208  text3 += " (" + key.toString(QKeySequence::NativeText) + ")";
209 
210  m_shortcut = new QShortcut(key, parent->getCanvas());
211  m_shortcut->setContext(context);
212  }
213 
214  m_action = new QAction(icon, text2, parent);
215  m_action->setToolTip(text3);
216  }

277  bool eventFilter(QObject *obj, QEvent *event)
278  {
279  if (event->type() == QEvent::Close)
280  {
281  closeEvent((QCloseEvent*)event);
282  if(testAttribute(Qt::WA_DeleteOnClose) && event->isAccepted() ) //if window should be closed and dockwidget is assigned with WA_DeleteOnClose, delete this dockwidget first.
283  {
284  deleteLater();
285  }
286 
287  return true;
288  }
289  else
290  {
291  return QObject::eventFilter(obj,event);
292  }
